(progn
  (save-excursion
    (goto-char (point-min))
    (while (re-search-forward ";;;+\n;; *\\(.*?\\) *;+\n;;;+" nil t)
      (replace-match ";;; \\1")))
  (save-excursion
    (goto-char (point-min))
    (while (re-search-forward "^;;; -- \\(.*\\) -+$" nil t)
      (replace-match ";;;; \\1"))))
